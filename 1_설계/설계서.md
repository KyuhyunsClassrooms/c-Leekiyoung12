## 3. 데이터 구조 설계
(프로그램에 필요한 핵심 변수, 배열을 설계하세요. 필요하면 구조체(struct)를 사용할 수도 있습니다.)


struct Student {
    char name[20];
    int score;
    int grade=1;
    
};
int n;
scanf("%d,&n);
struct Student students[n];  // 학생 n명 저장

for(int i=0;i<n;i++){
    scanf("%s",&students[i].name);
    scanf("%d",&students[i].score);
} //학생정보입력

for(int i=0;i<n;i++){
    for(int j=0;j<n;j++){
        if(students[i].score <students[j].score){
            students[i].grade++;
        }
    }
}//석차계산
